import sqlite3

def init_db():
    conn = sqlite3.connect('odontologia.db')
    cursor = conn.cursor()

    cursor.execute('''
    CREATE TABLE IF NOT EXISTS pacientes (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        nome TEXT NOT NULL,
        telefone TEXT NOT NULL,
        data_nascimento TEXT NOT NULL
    )
    ''')

    cursor.execute('''
    CREATE TABLE IF NOT EXISTS consultas (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        paciente_id INTEGER NOT NULL,
        data_hora TEXT NOT NULL UNIQUE,
        FOREIGN KEY (paciente_id) REFERENCES pacientes (id)
    )
    ''')

    conn.commit()
    conn.close()

def add_paciente(nome, telefone, data_nascimento):
    conn = sqlite3.connect('odontologia.db')
    cursor = conn.cursor()
    cursor.execute('''
    INSERT INTO pacientes (nome, telefone, data_nascimento)
    VALUES (?, ?, ?)
    ''', (nome, telefone, data_nascimento))
    conn.commit()
    paciente_id = cursor.lastrowid
    conn.close()
    return paciente_id

def add_consulta(paciente_id, data_hora):
    conn = sqlite3.connect('odontologia.db')
    cursor = conn.cursor()
    try:
        cursor.execute('''
        INSERT INTO consultas (paciente_id, data_hora)
        VALUES (?, ?)
        ''', (paciente_id, data_hora))
        conn.commit()
        conn.close()
        return True
    except sqlite3.IntegrityError:
        conn.close()
        return False

def horario_ocupado(data_hora):
    conn = sqlite3.connect('odontologia.db')
    cursor = conn.cursor()
    cursor.execute('SELECT 1 FROM consultas WHERE data_hora = ?', (data_hora,))
    existe = cursor.fetchone() is not None
    conn.close()
    return existe

def get_consultas():
    conn = sqlite3.connect('odontologia.db')
    cursor = conn.cursor()
    cursor.execute('''
    SELECT c.id, p.nome, p.telefone, c.data_hora
    FROM consultas c
    JOIN pacientes p ON c.paciente_id = p.id
    ORDER BY c.data_hora
    ''')
    consultas = cursor.fetchall()
    conn.close()
    return consultas

from telegram import Update, InlineKeyboardButton, InlineKeyboardMarkup
from telegram.ext import (
    Application, CommandHandler, MessageHandler, filters,
    CallbackContext, CallbackQueryHandler, ConversationHandler
)
from database import (
    init_db, add_paciente, add_consulta,
    horario_ocupado, get_consultas
)
import logging

logging.basicConfig(
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    level=logging.INFO
)
logger = logging.getLogger(__name__)

# Estados da conversa
NOME, TELEFONE, DATA_NASCIMENTO, DATA_HORA = range(4)

init_db()

async def start(update: Update, context: CallbackContext):
    keyboard = [
        [InlineKeyboardButton("Agendar Consulta", callback_data='agendar')],
        [InlineKeyboardButton("Listar Consultas", callback_data='listar')]
    ]
    reply_markup = InlineKeyboardMarkup(keyboard)
    await update.message.reply_text(
        "Bem-vindo ao Bot Odontológico!\nEscolha uma opção:",
        reply_markup=reply_markup
    )

async def button(update: Update, context: CallbackContext):
    query = update.callback_query
    await query.answer()
    if query.data == 'agendar':
        await query.edit_message_text("Qual é o seu nome?")
        return NOME
    elif query.data == 'listar':
        consultas = get_consultas()
        if not consultas:
            await query.edit_message_text("Nenhuma consulta agendada.")
        else:
            texto = "📅 Consultas Agendadas:\n\n"
            for c in consultas:
                texto += f"ID: {c[0]}\nPaciente: {c[1]}\nTelefone: {c[2]}\nData/Hora: {c[3]}\n\n"
            await query.edit_message_text(texto)
        return ConversationHandler.END

async def nome(update: Update, context: CallbackContext):
    context.user_data['nome'] = update.message.text
    await update.message.reply_text("Qual é o seu telefone?")
    return TELEFONE

async def telefone(update: Update, context: CallbackContext):
    context.user_data['telefone'] = update.message.text
    await update.message.reply_text("Qual é a sua data de nascimento? (DD/MM/AAAA)")
    return DATA_NASCIMENTO

async def data_nascimento(update: Update, context: CallbackContext):
    context.user_data['data_nascimento'] = update.message.text
    await update.message.reply_text("Qual é a data e hora da consulta? (DD/MM/AAAA HH:MM)")
    return DATA_HORA

async def data_hora(update: Update, context: CallbackContext):
    data_hora = update.message.text
    if horario_ocupado(data_hora):
        await update.message.reply_text("Esse horário já está ocupado. Escolha outro.")
        return DATA_HORA
    paciente_id = add_paciente(
        context.user_data['nome'],
        context.user_data['telefone'],
        context.user_data['data_nascimento']
    )
    if add_consulta(paciente_id, data_hora):
        await update.message.reply_text(
            f"✅ Consulta agendada!\n"
            f"Paciente: {context.user_data['nome']}\n"
            f"Telefone: {context.user_data['telefone']}\n"
            f"Data/Hora: {data_hora}"
        )
    else:
        await update.message.reply_text("Erro ao agendar consulta. Tente novamente.")
    return ConversationHandler.END

async def cancel(update: Update, context: CallbackContext):
    await update.message.reply_text("Operação cancelada.")
    return ConversationHandler.END

def main():
    # Substitua pelo token do seu bot
    application = Application.builder().token("SEU_TOKEN_AQUI").build()

    conv_handler = ConversationHandler(
        entry_points=[CommandHandler('start', start)],
        states={
            NOME: [MessageHandler(filters.TEXT & ~filters.COMMAND, nome)],
            TELEFONE: [MessageHandler(filters.TEXT & ~filters.COMMAND, telefone)],
            DATA_NASCIMENTO: [MessageHandler(filters.TEXT & ~filters.COMMAND, data_nascimento)],
            DATA_HORA: [MessageHandler(filters.TEXT & ~filters.COMMAND, data_hora)],
        },
        fallbacks=[CommandHandler('cancel', cancel)],
    )

    application.add_handler(CommandHandler('start', start))
    application.add_handler(CallbackQueryHandler(button))
    application.add_handler(conv_handler)

    application.run_polling()

if __name__ == '__main__':
    main()
